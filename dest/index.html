Easy Button Group

<div class="btn-group">
  <button ng-class="{active: states.left}" ng-click="states.$update({left: !states.left})" class="btn">Left</button>
  <button ng-class="{active: states.middle}" ng-click="states.$update({middle: !states.middle})" class="btn">Middle</button>
  <button ng-class="{active: states.right}" ng-click="states.$update({right: !states.right})" class="btn">Right</button>
</div>
 
this.ButtonGroupCtrl = ['$log', '$scope', 'FireSync'].concat(function($log, $scope, FireSync){
  $scope.states = new FireSync().get(FirebaseURL + "/button-states").sync();
  $log.log('ButtonGroupCtrl', $scope);
});

Easy Button Toolbar

<div class="btn-toolbar">
  <div class="btn-group">
    <button ng-repeat="state in states track by state.$name" ng-class="{active: state.$value}" ng-click="state.$set(!state.$value)" class="btn">
       
      {{ state.$name }}
    </button>
  </div>
</div>
 
this.ButtonToolbarCtrl = ['$log', '$scope', 'FireSync'].concat(function($log, $scope, FireSync){
  $scope.states = new FireSync().get(FirebaseURL + "/button-states", {
    toCollection: true
  }).sync();
  $log.log('ButtonToolbarCtrl', $scope);
});
